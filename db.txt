-- public.concept definition

-- Drop table

-- DROP TABLE public.concept;

CREATE TABLE public.concept (
	concept_id uuid NOT NULL, -- 개념 고유 ID(UUID)
	pdf_id uuid NOT NULL, -- 이 개념이 속한 PDF의 고유 ID
	title varchar(255) NOT NULL, -- 개념 제목
	description text NULL, -- 개념에 대한 짧은 설명
	start_page int4 NOT NULL, -- 개념이 시작되는 PDF 페이지 번호
	end_page int4 NOT NULL, -- 개념이 끝나는 PDF 페이지 번호
	order_index int4 NOT NULL, -- 같은 PDF 또는 같은 부모 개념 아래에서 표시 순서
	concept_type varchar(30) NULL, -- 개념 유형(intro, section, concept, subconcept, example, summary)
	hierarchy_level int4 DEFAULT 0 NOT NULL, -- 문서 구조상 깊이 수준(0=최상위, 1=하위, 2=더 하위...)
	parent_concept_id uuid NULL, -- 상위 개념의 ID, 최상위 개념이면 NULL
	path_text text NULL, -- 개념의 전체 경로 문자열 예: 협업 필터링 > 메모리 기반 CF > User-based CF
	CONSTRAINT chk_concept_hierarchy_level CHECK ((hierarchy_level >= 0)),
	CONSTRAINT chk_concept_page_range CHECK ((start_page <= end_page)),
	CONSTRAINT chk_concept_type CHECK (((concept_type IS NULL) OR ((concept_type)::text = ANY ((ARRAY['intro'::character varying, 'section'::character varying, 'concept'::character varying, 'subconcept'::character varying, 'example'::character varying, 'summary'::character varying])::text[])))),
	CONSTRAINT concept_pkey PRIMARY KEY (concept_id),
	CONSTRAINT concept_pdf_id_fkey FOREIGN KEY (pdf_id) REFERENCES public.pdf(pdf_id),
	CONSTRAINT fk_concept_parent FOREIGN KEY (parent_concept_id) REFERENCES public.concept(concept_id) ON DELETE SET NULL
);
COMMENT ON TABLE public.concept IS 'PDF 구조 분석 결과에서 추출된 개념 정보를 저장하는 테이블';

-- Column comments

COMMENT ON COLUMN public.concept.concept_id IS '개념 고유 ID(UUID)';
COMMENT ON COLUMN public.concept.pdf_id IS '이 개념이 속한 PDF의 고유 ID';
COMMENT ON COLUMN public.concept.title IS '개념 제목';
COMMENT ON COLUMN public.concept.description IS '개념에 대한 짧은 설명';
COMMENT ON COLUMN public.concept.start_page IS '개념이 시작되는 PDF 페이지 번호';
COMMENT ON COLUMN public.concept.end_page IS '개념이 끝나는 PDF 페이지 번호';
COMMENT ON COLUMN public.concept.order_index IS '같은 PDF 또는 같은 부모 개념 아래에서 표시 순서';
COMMENT ON COLUMN public.concept.concept_type IS '개념 유형(intro, section, concept, subconcept, example, summary)';
COMMENT ON COLUMN public.concept.hierarchy_level IS '문서 구조상 깊이 수준(0=최상위, 1=하위, 2=더 하위...)';
COMMENT ON COLUMN public.concept.parent_concept_id IS '상위 개념의 ID, 최상위 개념이면 NULL';
COMMENT ON COLUMN public.concept.path_text IS '개념의 전체 경로 문자열 예: 협업 필터링 > 메모리 기반 CF > User-based CF';

-- Constraint comments

COMMENT ON CONSTRAINT chk_concept_hierarchy_level ON public.concept IS '계층 레벨은 0 이상만 허용';
COMMENT ON CONSTRAINT chk_concept_page_range ON public.concept IS '시작 페이지가 종료 페이지보다 클 수 없도록 제한';
COMMENT ON CONSTRAINT chk_concept_type ON public.concept IS '허용된 개념 유형만 저장되도록 제한';
COMMENT ON CONSTRAINT fk_concept_parent ON public.concept IS '상위 개념을 같은 concept 테이블에서 참조하는 자기참조 외래키';
